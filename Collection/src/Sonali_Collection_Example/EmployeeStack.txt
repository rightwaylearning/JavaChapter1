package Soanli_Colletion;
import java.util.Iterator;
import java.util.LinkedList;
public class EmployeeStack {
	
	 LinkedList<EmployeeInfo> stack;
	 
	 EmployeeStack() {
			this.stack = new LinkedList<EmployeeInfo>();
		}
	 
	 public void push(EmployeeInfo data) {
			if (data != null) {
				this.stack.addFirst(data);
			}
		}

		public EmployeeInfo pop() {
			if (this.stack.isEmpty()) {
				System.out.println("Sorry!! Stack is empty, please push first.");
				return null;
			} else {
				return this.stack.removeFirst();
			}
		}
	 public void viewStack() {
			Iterator<EmployeeInfo> itr = this.stack.iterator();
			while(itr.hasNext()) {
				System.out.println(itr.next());
			}
		}

	}
